{{- /* JSON-LD structured data: BlogPosting on posts, WebSite + Person on home. */ -}} {{- /* Standalone pages like /taste and /places get neither: they are not */ -}} {{- /* articles, and BlogPosting would emit their zero dates as 0001-01-01. */ -}} {{- $author := .Site.Params.author.name | title -}} {{- if and .IsPage (eq .Section "blog") -}} {{- $img := "" -}} {{- with partial "og-image.html" . -}} {{ $img = .Permalink }} {{- else -}} {{- with .Site.Params.images -}}{{ $img = (index . 0) | absURL }}{{- end -}} {{- end -}} {{- $data := dict "@context" "https://schema.org" "@type" "BlogPosting" "headline" .Title "url" .Permalink "datePublished" (.Date.Format "2006-01-02T15:04:05Z07:00") "dateModified" (.Lastmod.Format "2006-01-02T15:04:05Z07:00") "author" (dict "@type" "Person" "name" $author "url" site.Home.Permalink) "publisher" (dict "@type" "Person" "name" $author) "mainEntityOfPage" (dict "@type" "WebPage" "@id" .Permalink) -}} {{- with .Description | default .Summary -}} {{ $data = merge $data (dict "description" .) }} {{- end -}} {{- with $img -}}{{ $data = merge $data (dict "image" .) }}{{- end -}} {{- else if .IsHome -}} {{- $data := dict "@context" "https://schema.org" "@type" "WebSite" "name" .Site.Title "url" site.Home.Permalink "description" .Site.Params.description "author" (dict "@type" "Person" "name" $author "url" site.Home.Permalink) -}} {{- end -}}
